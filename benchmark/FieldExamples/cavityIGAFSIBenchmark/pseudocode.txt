/*

 * Pseudo code of the server (EMPEROR)
 * ================================================
 */
int main() {
	// Stage 1: receive meshes ...
	receiveMeshFrom("carat", "myMesh1");
	receiveMeshFrom("OpenFOAM", "myMesh1");
	// Stage 2: do co-simulation ...
	//  ...
	//  ...
	for (int i=0; i<numberOfTimeSteps; i++) {// time step loop
		//  ...
		while (!isConvergent) { // iterative coupling loop
			//  ...
			sendDataFieldTo("OpenFOAM", "myMesh1", "displacements");
			//  ...
			//  ...
			receiveDataFieldFrom("OpenFOAM", "myMesh1", "tractionsElem");
			sendDataFieldTo("carat", "myMesh1", "forces");
			//  ...
			//  ...
			receiveDataFieldFrom("carat", "myMesh1", "displacements");
			//  ...
			sendConvergenceSignalTo(carat, isConvergent);
			sendConvergenceSignalTo(OpenFOAM, isConvergent);
		}
	}
	//  ...
	return;
}
